#include<bits/stdc++.h>

using namespace std;

int main()
{
    int n1,n2,cnt=0,i,j,k,d,n,res;
    cin>>n>>d>>n1;
    for(i=1;i<n;i++)
    {
        cin>>n2;
        if(n1>=n2)
        {
         res=(n1-n2)/d+1;
         cnt+=res;
         n1=n2+res*d;
        }
        else
            n1=n2;
    }

    cout<<cnt<<endl;

    return 0;
}
